```
static bool Initialized = false;
if (!Initialized) {
    player.Initialize();
    Initialized = true;
}

player.Update(g_pd3dDevice);

ImGui::Begin("Window");
{
    if (player.HasMedia)
    {
        ImGui::Text(player.Title);
        ImGui::Text(player.Artist);
        if (player.thumb != nullptr)
            ImGui::Image(player.thumb, ImVec2(30, 30));
    }
}
ImGui::End();
 ```
